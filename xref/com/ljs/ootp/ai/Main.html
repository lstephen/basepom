268                 PitchingRatings<?> vL = rs.getVsLeft();
269                 PitchingRatings<?> vR = rs.getVsRight();

271                 base = (vL.getStuff() + vL.getMovement() + vL.getControl()
272                      + 2.0 * (vR.getStuff() + vR.getMovement() + vR.getControl())) / 3.0;
273               } else {

275

277                 BattingRatings<?> vR = rs.getVsRight();
278 
279                 base = (vL.getContact() + vL.getPower() + vL.getEye()
280                      + 2.0 * (vR.getContact() + vR.getPower() + vR.getEye())) / 3.0;
281               }

283               Integer need = Slot.getPlayerSlots(p)
284                 .stream()
285                 .filter((s) -> s != Slot.P)
286                 .filter((s) -> s != Slot.H || Slot.getPrimarySlot(p) == Slot.H)
287                 .map((s) -> rr.getTargetRatio() - rr.getRatio(s))
288                 .max(Integer::compare)
289                 .orElse(0);

291 
292               need = Math.max(0, need);
293 
294               int intangibles = 0;
295 
296               if (p.getClutch().isPresent()) {
297                 switch (p.getClutch().get()) {
298                   case GREAT:
299                     intangibles += 1;
300                     break;
301                   case SUFFERS:
302                     intangibles += -1;
303                     break;
304                   default:
305                     // do nothing
306                 }
307               }
308 
309               if (p.getConsistency().isPresent()) {
310                 switch (p.getConsistency().get()) {
311                   case VERY_INCONSISTENT:
312                     intangibles -= 1;
313                     break;
314                   case GOOD:
315                     intangibles += 1;
316                     break;
317                   default:
318                     // do nothing
319                 }
320               }
321 
322               Double mrFactor = Slot.getPrimarySlot(p) == Slot.MR ? 0.865 : 1.0;
323 
324               return new Double(mrFactor * base - p.getAge() + need - PlayerValue.getAgingFactor(p) + intangibles).intValue();
325             });
326           }
