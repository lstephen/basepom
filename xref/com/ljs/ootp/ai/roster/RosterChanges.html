19  public class RosterChanges implements Printable {
20  
21      private final Set<RosterChange> changes = Sets.newHashSet();
22  
23      public void addChange(Player player, Status from, Status to) {
24          changes.add(new RosterChange(player, from, to));
25      }
26  
27      @Override
28      public void print(PrintWriter w) {
29          w.println();
30          for (RosterChange c : RosterChange.ordering().sortedCopy(changes)) {
31              c.println(w);
32          }
33      }
34  
35      private static class RosterChange {
36          private final Player player;
37          private final Status from;
38          private final Status to;
39  
40          public RosterChange(Player player, Status from, Status to) {
41              this.player = player;
42              this.from = from;
43              this.to = to;
44          }
45  
46          public void println(PrintWriter w) {
47              w.println(
48                  String.format(
49                      "%4s -> %-4s %2s %s",
50                      from == null ? "" : from,
51                      to == null ? "" : to,
52                      player.getPosition(),
53                      player.getShortName()));
54          }
55  
56          public static Ordering<RosterChange> ordering() {
57              return Ordering
58                  .natural()
59                  .nullsLast()
60                  .onResultOf((RosterChange c) -> c.from)
61                  .compound(Ordering
62                      .natural()
63                      .nullsLast()
64                      .onResultOf(c -> c.to));
65          }
66      }

69  }
