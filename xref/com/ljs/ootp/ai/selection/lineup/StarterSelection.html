68      private Player selectDh(Lineup.VsHand vs, Iterable<Player> available) {
69          if (Iterables.size(available) == 1) {
70              return available.iterator().next();
71          }
72  
73          for (Player p : byWoba(vs).sortedCopy(available)) {
74              if (!requireBackupCatcher) {

76              } else {
77                  if ((!p.getSlots().contains(Slot.C))

79                          Sets.difference(ImmutableSet.copyOf(available),
