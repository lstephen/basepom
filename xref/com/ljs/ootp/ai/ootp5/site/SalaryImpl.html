34      public SalaryImpl(Site site, Id<Team> team) {
35          Preconditions.checkNotNull(site);
36          Preconditions.checkNotNull(team);
37  
38          this.site = site;
39          this.team = team;
40      }

42      private Page getPage() {
43          return site.getPage("team" + team.get() + "sa.html");
44      }
45  
46      private Document loadPage() {
47          return getPage().load();
48      }

50      public String getSalary(Player p) {
51          Document doc = loadPage();
52  
53          for (Element el : doc.select("tr.g:has(a), tr.g2:has(a)")) {
54              Element a = el.children().select("a").get(0);
55              if (p.getId().equals(new PlayerId(a.attr("href").replaceAll(".html", "")))) {
56                  String salary = el.children().get(2).text().trim();
57  
58                  if (salary.startsWith("$")) {
59                      String value;
60                      try {
61                          value = SalaryFormat.prettyPrint(NumberFormat.getNumberInstance().parse(salary.substring(1)));
62                      } catch (ParseException e) {
63                          throw Throwables.propagate(e);
64                      }
65  
66                      char years = el.children().get(4).text().trim().charAt(0);
67  
68                      if (years != '1') {
69                          return value + "x" + years;
70                      } else {
71                          String comment = el.children().get(5).text();
72                          if (comment.contains("Extension")) {
73                              return value + " e";
74                          } else if (comment.contains("Automatic")) {
75                              return value + " r";
76                          } else if (comment.contains("Arbitration")) {
77                              return value + " a";
78                          } else {
79                              return value + "  ";
80                          }
81                      }
82                  } else {
83                      return "";
84                  }
85              }
86          }
87          return null;
88      }

90      public Integer getCurrentSalary(final Player p) {
91          String key = site.getName() + p.getId();
92  
93          if (CURRENT_SALARY_CACHE.containsKey(key)) {
94              return CURRENT_SALARY_CACHE.get(key);
95          }
96  
97          Document doc = loadPage();
98  
99          for (Element el : doc.select("tr.g:has(a), tr.g2:has(a)")) {
100             Element a = el.children().select("a").get(0);
101             if (p.getId().equals(new PlayerId(a.attr("href").replaceAll(".html", "")))) {
102                 String salary = el.children().get(2).text().trim();
103 
104                 if (salary.startsWith("$")) {
105                     try {
106                         Integer result = NumberFormat.getIntegerInstance().parse(salary.substring(1)).intValue();
107 
108                         CURRENT_SALARY_CACHE.put(key, result);
109 
110                         return result;
111                     } catch (ParseException e) {
112                         throw Throwables.propagate(e);
113                     }
114                 }
115             }
116         }
117 
118         return 0;
119     }

121     public Integer getNextSalary(Player p) {
122         Document doc = loadPage();
123 
124         for (Element el : doc.select("tr.g:has(a), tr.g2:has(a)")) {
125             Element a = el.children().select("a").get(0);
126             if (p.getId().equals(new PlayerId(a.attr("href").replaceAll(".html", "")))) {
127                 String salary = el.children().get(2).text().trim();
128 
129                 if (salary.startsWith("$")) {
130                     char years = el.children().get(4).text().trim().charAt(0);
131                     if (years != '1') {
132                         try {
133                             return NumberFormat.getIntegerInstance().parse(salary.substring(1)).intValue();
134                         } catch (ParseException e) {
135                             throw Throwables.propagate(e);
136                         }
137                     } else {
138                         String comment = el.children().get(5).text().trim();
139 
140                         if (comment.contains("Automatic") || comment.contains("Possible Arbitration")) {
141                             return getCurrentSalary(p);
142                         } else if (comment.contains("Arbitration (Estimate:")) {
143                             try {
144                                 int estimate = NumberFormat.getIntegerInstance().parse(
145                                     StringUtils.substringBetween(comment, "$", ")")).intValue();
146 
147                                 return Math.max(getCurrentSalary(p), estimate);
148                             } catch (ParseException e) {
149                                 throw Throwables.propagate(e);
150                             }
151                         }
152                     }
153                 }
154             }
155         }
156 
157         return 0;
158     }
