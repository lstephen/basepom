3   import org.apache.commons.lang3.StringUtils;
4   import org.jsoup.nodes.Document;
5   
6   public enum Clutch {
7     SUFFERS, NORMAL, GREAT;
8   
9     public static Clutch extractFrom(Document doc) {
10        String starsText = doc.select("td.s4:containsOwn(Stars)").html();
11        //System.out.println(starsText);
12        String[] split = StringUtils.splitByWholeSeparator(starsText, "<br />");
13        //System.out.println(Strings.join(split).with(","));
14        String clutchText = split[split.length - 4];
15  
16        if (clutchText.equalsIgnoreCase("Suffers")) {
17          return SUFFERS;
18        } else if (clutchText.equalsIgnoreCase("Normal")) {
19          return NORMAL;
20        } else if (clutchText.equalsIgnoreCase("GREAT")) {
21          return GREAT;
22        }
23  
24        throw new IllegalArgumentException("Unknown Clutch Rating: " + clutchText);
25    }
26  }
