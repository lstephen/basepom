3   import com.google.common.base.Function;
4   import com.google.common.collect.Ordering;
5   import com.ljs.ootp.ai.io.Printable;
6   import com.ljs.ootp.ai.player.Player;
7   import com.ljs.ootp.ai.roster.Team;
8   import com.ljs.ootp.ai.site.Salary;
9   import com.ljs.ootp.ai.site.Site;
10  import java.io.PrintWriter;
11  import java.text.NumberFormat;
12  import javax.annotation.ParametersAreNonnullByDefault;
13  import org.apache.commons.lang3.StringUtils;
14  
15  /**
16   *
17   * @author lstephen
18   */
19  @ParametersAreNonnullByDefault
20  public class SalaryReport implements Printable {
21  
22      private final Team team;
23  
24      private final Salary salary;
25  
26      private Integer total;
27  
28      public SalaryReport(Team team, Site site) {
29          this.team = team;
30          this.salary = site.getSalary();
31      }
32  
33      public Integer getCurrentSalary(Player p) {
34          return salary.getCurrentSalary(p);
35      }
36  
37      public Integer getCurrentTotal() {
38          if (total == null) {
39              total = 0;
40  
41              for (Player p : team) {
42                  total += salary.getCurrentSalary(p);
43              }
44          }
45  
46          return total;
47      }
48  
49      @Override
50      public void print(PrintWriter w) {
51          w.println();
52  
53          int nextTotal = 0;
54  
55          for (Player p : byCurrentSalary().reverse().sortedCopy(team)) {
56              Integer s = salary.getCurrentSalary(p);
57  
58              if (s == 0) {
59                  continue;
60              }
61  
62              Integer nextS = salary.getNextSalary(p);
63  
64              nextTotal += nextS;
65  
66              w.format(
67                  "%2s %-15s %2d| %11s %11s%n",
68                  p.getPosition(),
69                  StringUtils.abbreviate(p.getShortName(), 15),
70                  p.getAge(),
71                  NumberFormat.getIntegerInstance().format(s),
72                  nextS == 0
73                      ? ""
74                      : NumberFormat.getIntegerInstance().format(nextS));
75          }
76  
77          w.println(
78              String.format(
79                  "%21s| %11s %11s",
80                  "",
81                  NumberFormat.getIntegerInstance().format(getCurrentTotal()),
82                  NumberFormat.getIntegerInstance().format(nextTotal)));
83  
84          w.flush();
85      }
86  
87      private Ordering<Player> byCurrentSalary() {
88          return Ordering
89              .natural()
90              .onResultOf(new Function<Player, Integer>() {
91                  @Override
92                  public Integer apply(Player p) {
93                      return salary.getCurrentSalary(p);
94                  }
95              });
96      }
97  
98  }
