4   import com.google.common.base.Predicate;

15  /**
16   *
17   * @author lstephen
18   */
19  public final class RosterReport implements Printable {
20  
21      private final Iterable<Player> roster;
22  
23      private Integer targetRatio = 40;
24  
25      private RosterReport(Iterable<Player> roster) {
26          this.roster = roster;
27      }
28  
29      public Integer getTargetRatio() {
30          return targetRatio;
31      }
32  
33      private void setTargetRatio(Integer ratio) {
34          this.targetRatio = ratio;
35      }
36  
37      private int getPrimaryCount(Slot s) {
38          int count = 0;
39          for (Player p : roster) {
40              if (Slot.getPrimarySlot(p) == s) {
41                  count++;
42              }
43          }
44          return count;
45      }
46  
47      private int getAnyCount(Slot s) {
48          int count = 0;
49          for (Player p : roster) {
50              if (Slot.getPlayerSlots(p).contains(s)) {
51                  count++;
52              }
53          }
54          return count;
55      }
56  
57      public Integer getRatio(Slot s) {
58          return getPrimaryCount(s) * 10
59              / Math.max(
60                  Mode.REGULAR_SEASON.getHittingSlots().count(s),
61                  Slot.MAJOR_LEAGUE_PITCHING_SLOTS.count(s));
62      }
63  
64      public Set<Slot> getSurplusSlots() {
65          Set<Slot> slots = Sets.newHashSet(Slot.values());
66  
67          slots.remove(Slot.P);
68  
69          return Sets.filter(slots, new Predicate<Slot>() {
70              public boolean apply(Slot s) {
71                  return getRatio(s) > (targetRatio * 1.15);
72              }
73          });
74      }
75  
76      public Set<Slot> getNeededSlots() {
77          Set<Slot> slots = Sets.newHashSet(Slot.values());
78  
79          slots.remove(Slot.P);
80  
81          return Sets.filter(slots, new Predicate<Slot>() {
82              public boolean apply(Slot s) {
83                  return getRatio(s) < (targetRatio * .85);
84              }
85          });
86      }
87  
88      @Override
89      public void print(PrintWriter w) {
90          w.println();
91  
92          w.println("Roster Report");
93          w.print("Primary:");
94          for (Slot s : Slot.values()) {
95              if (s == Slot.P) {
96                  continue;
97              }
98              w.print(String.format(" %s:%3d ", s, getPrimaryCount(s)));
99          }
100         w.println();
101 
102         w.print("Any:    ");
103         for (Slot s : Slot.values()) {
104             if (s == Slot.P) {
105                 continue;
106             }
107             w.print(String.format(" %s:%3d ", s, getAnyCount(s)));
108         }
109         w.println();
110 
111         w.print("Ratio:  ");
112         for (Slot s : Slot.values()) {
113             if (s == Slot.P) {
114                 continue;
115             }
116             w.print(String.format(" %s:%3d ", s, getRatio(s)));
117         }
118         w.println();
119 
120         w.print("Needs:   ");
121         w.print(Joiner.on(',').join(getNeededSlots()));
122         w.println();
123 
124         w.print("Surplus: ");
125         w.print(Joiner.on(',').join(getSurplusSlots()));
126         w.println();
127     }
128 
129     public static RosterReport create(Site site, Roster roster) {
130         return create(site, roster.getAllPlayers());
131     }
132 
133     public static RosterReport create(Site site, Iterable<Player> ps) {
134         RosterReport report = new RosterReport(ps);
135 
136         if (site.getName().equals("PSD")) {
137             report.setTargetRatio(60);
138         }
139 
140         return report;
141     }
