17  public class LeagueBattingReport implements Printable {

19    private final BattingStats stats;

21    private final Double bsrFactor;
22  
23    private LeagueBattingReport(BattingStats stats) {
24      this.stats = stats;
25  
26      Double a = (double) stats.getHits() + stats.getWalks() - stats.getHomeRuns();
27  
28      Double b =
29        BaseRuns.COEFFICIENT_SINGLE * stats.getSingles() +
30        BaseRuns.COEFFICIENT_DOUBLE * stats.getDoubles() +
31        BaseRuns.COEFFICIENT_TRIPLE * stats.getTriples() +
32        BaseRuns.COEFFICIENT_HOME_RUN * stats.getHomeRuns() +
33        BaseRuns.COEFFICIENT_WALK * stats.getWalks() +
34        BaseRuns.COEFFICIENT_STRIKEOUT * stats.getStrikeouts() +
35        BaseRuns.COEFFICIENT_OUT * (stats.getOuts() - stats.getStrikeouts());
36  
37      Double c = (double) stats.getOuts();
38  
39      Double d = (double) stats.getHomeRuns();
40  
41      Double z = (stats.getRuns() - d) / a;
42      bsrFactor = (z * c/(1 - z)) / b;
43    }
44  
45    @Override
46    public void print(PrintWriter w) {
47  
48      w.format("bsf: %.2f%n", bsrFactor);
49      w.format("act: %d%n", stats.getRuns());
50      w.println();
51  
52      Double runBB = getRunValueBB();
53      Double run1B = getRunValue1B();
54      Double run2B = getRunValue2B();
55      Double run3B = getRunValue3B();
56      Double runHR = getRunValueHR();
57  
58      w.println();
59      w.format("runBB: %.2f%n", runBB);
60      w.format("run1B: %.2f%n", run1B);
61      w.format("run2B: %.2f%n", run2B);
62      w.format("run3B: %.2f%n", run3B);
63      w.format("runHR: %.2f%n", runHR);
64  
65      Double runsMinus = (runBB * stats.getWalks()
66          + run1B * stats.getSingles()
67          + run2B * stats.getDoubles()
68          + run3B * stats.getTriples()
69          + runHR * stats.getHomeRuns())
70        / stats.getOuts();
71  
72      Double runsPlus = (runBB * stats.getWalks()
73          + run1B * stats.getSingles()
74          + run2B * stats.getDoubles()
75          + run3B * stats.getTriples()
76          + runHR * stats.getHomeRuns())
77        / (stats.getWalks() + stats.getHits());
78  
79      Double wobaScale = 1 / (runsPlus + runsMinus);
80  
81      Double bipValue = stats.getBabip() * runsPlus - (1.0 - stats.getBabip()) * runsMinus;
82  
83      w.format("rnBIP: %.2f%n", bipValue);
84  
85      w.println();
86      w.format("runs-: %.2f%n", runsMinus);
87      w.format("runs+: %.2f%n", runsPlus);
88      w.format("wobas: %.2f%n", wobaScale);
89  
90  
91      Double wobaBB = (runBB + runsMinus) * wobaScale;
92      Double woba1B = (run1B + runsMinus) * wobaScale;
93      Double woba2B = (run2B + runsMinus) * wobaScale;
94      Double woba3B = (run3B + runsMinus) * wobaScale;
95      Double wobaHR = (runHR + runsMinus) * wobaScale;
96  
97      w.println();
98      w.format("wobaBB: %.2f%n", wobaBB);
99      w.format("woba1B: %.2f%n", woba1B);
100     w.format("woba2B: %.2f%n", woba2B);
101     w.format("woba3B: %.2f%n", woba3B);
102     w.format("wobaHR: %.2f%n", wobaHR);
103 
104     Woba.setConstants(ImmutableMap
105         .<Woba.Event, Double>builder()
106         .put(Woba.Event.WALK, wobaBB)
107         .put(Woba.Event.SINGLE, woba1B)
108         .put(Woba.Event.DOUBLE, woba2B)
109         .put(Woba.Event.TRIPLE, woba3B)
110         .put(Woba.Event.HOME_RUN, wobaHR)
111         .build());
112 
113     w.println();
114 
115     FipBaseRuns.setFactor(bsrFactor);
116     FipBaseRuns.setLeagueContext(stats);
117     EraBaseRuns.setFactor(bsrFactor);
118     EraBaseRuns.setLeagueContext(stats);
119   }

121   private Double getRunValueBB() {
122     BattingStats plusOne = new BattingStats();
123     plusOne.setWalks(1);
124 
125     return baseRuns(stats.add(plusOne)) - stats.getRuns();
126   }

164   private Double baseRuns(BattingStats stats) {
165 
166     Double a = (double) stats.getHits() + stats.getWalks() - stats.getHomeRuns();
167 
168     Double b =
169       BaseRuns.COEFFICIENT_SINGLE * stats.getSingles() +
170       BaseRuns.COEFFICIENT_DOUBLE * stats.getDoubles() +
171       BaseRuns.COEFFICIENT_TRIPLE * stats.getTriples() +
172       BaseRuns.COEFFICIENT_HOME_RUN * stats.getHomeRuns() +
173       BaseRuns.COEFFICIENT_WALK * stats.getWalks() +
174       BaseRuns.COEFFICIENT_STRIKEOUT * stats.getStrikeouts() +
175       BaseRuns.COEFFICIENT_OUT * (stats.getOuts() - stats.getStrikeouts());
176 
177     Double c = (double) stats.getOuts();
178 
179     Double d = (double) stats.getHomeRuns();
180 
181     b = bsrFactor * b;
182 
183     return a * b/(b+c) + d;
184   }
