24  public class SelectedPlayers {
25  
26      private final ImmutableSet<Player> players;
27  
28      private final TeamStats<BattingStats> predictions;
29  
30      private final SplitPercentages splits;
31  
32      private SelectedPlayers(Iterable<Player> players, TeamStats<BattingStats> predictions, SplitPercentages splits) {
33          this.players = ImmutableSet.copyOf(players);
34          this.predictions = predictions;
35          this.splits = splits;
36      }
37  
38      public ImmutableMultimap<Slot, Player> selection() {
39          Multimap<Slot, Player> sel = HashMultimap.create();
40  
41          for (Player p : players) {
42              sel.put(Slot.getPrimarySlot(p), p);
43          }
44  
45          return ImmutableMultimap.copyOf(sel);
46      }
47  
48      public ImmutableSet<Player> players() {
49          return players;
50      }
51  
52      public Boolean contains(Player p) {
53          return players.contains(p);
54      }
55  
56      public Double score() {
57          AllLineups lineups = new LineupSelection(predictions).select(players);
58  
59          return splits.getVsRhpPercentage() * score(VsHand.VS_RHP, lineups.getVsRhpPlusDh())
60              + splits.getVsRhpPercentage() * score(VsHand.VS_RHP, lineups.getVsRhp())
61              + splits.getVsLhpPercentage() * score(VsHand.VS_LHP, lineups.getVsLhpPlusDh())
62              + splits.getVsLhpPercentage() * score(VsHand.VS_LHP, lineups.getVsLhp())
63              - players.size()
64              - (double) ageScore() / 100000;
65      }
66  
67      private Double score(Lineup.VsHand vs, Lineup lineup) {
68          return hittingScore(vs, lineup) + hittingWithDefenseScore(vs, lineup) + benchScore(vs, lineup);
69      }
70  
71      private Integer hittingScore(Lineup.VsHand vs, Lineup lineup) {
72          Integer score = 0;
73          for (Player p : lineup.playerSet()) {
74              score += vs.getStats(predictions, p).getWobaPlus();
75          }
76          return score;
77      }
78  
79      private Integer hittingWithDefenseScore(Lineup.VsHand vs, Lineup lineup) {
80          Integer score = 0;
81          for (Lineup.Entry entry : lineup) {
82              Position pos = entry.getPositionEnum();
83              if (pos == Position.PITCHER) {
84                  continue;
85              }
86              if (entry.getPlayer().canPlay(entry.getPositionEnum())) {
87                  score += vs.getStats(predictions, entry.getPlayer()).getWobaPlus();
88              }
89          }
90          return score;
91      }
92  
93      private Double benchScore(Lineup.VsHand vs, Lineup lineup) {
94          return new BenchScorer(predictions).score(Sets.difference(players, lineup.playerSet()), lineup, vs);
95      }
96  
97      private Integer overallHittingScore() {
98          Integer score = 0;
99          for (Player p : players) {
100             // This already combines according to split percentages
101             score += predictions.getSplits(p).getOverall().getWobaPlus();
102         }
103         return score;
104     }
105 
106     private Integer ageScore() {
107         Integer score = 0;
108         for (Player p : players) {
109             score += p.getAge();
110         }
111         return score;
112     }
113 
114     public static SelectedPlayers create(Iterable<Player> ps, TeamStats<BattingStats> predictions, SplitPercentages pcts) {
115         return new SelectedPlayers(ps, predictions, pcts);
116     }
117 
118 }
