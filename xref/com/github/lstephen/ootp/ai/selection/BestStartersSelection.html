57      private Integer getTargetSize() {
58          return slots.size();
59      }

61  	@Override
62  	public ImmutableMultimap<Slot, Player> select(Iterable<Player> forced, Iterable<Player> available) {
63  
64          ImmutableSet<Player> best = ImmutableSet.copyOf(Iterables.concat(selectStarters(available), forced));
65  
66          System.out.println("Best:");
67          for (Player p : Player.byShortName().sortedCopy(best)) {
68              System.out.print(p.getShortName() + "/");
69          }
70          System.out.println();
71  
72  
73          best = optimize(best, forced, available);
74  
75          Multimap<Slot, Player> result = HashMultimap.create();
76  
77          for (Player p : best) {
78              result.put(Slot.getPrimarySlot(p), p);
79          }
80  
81          return ImmutableMultimap.copyOf(result);
82  	}

84      private ImmutableSet<Player> optimize(Iterable<Player> best, Iterable<Player> forced, Iterable<Player> available) {
85          return optimize(ImmutableSet.copyOf(best), ImmutableSet.copyOf(forced), ImmutableSet.copyOf(available));
86  
87      }

89      private ImmutableSet<Player> optimize(ImmutableSet<Player> best, ImmutableSet<Player> forced, ImmutableSet<Player> available) {
90  
91          if (forced.size() >= getTargetSize()) {
92              return forced;
93          }
94  
95          if (best.size() > getTargetSize()) {
96              return optimize(limit(best, forced, getTargetSize()), forced, available);
97          }
98  
99          Double bestScore = 0.0;
100         ImmutableSet<Player> bestPlayers = null;
101 
102         Integer limit = best.size();
103 
104         while (limit >= 9) {
105             ImmutableSet<Player> ps = limit(best, forced, limit);
106 
107             ps = fill(ps, available);
108             System.out.print("Filled: ");
109             for (Player p : Player.byShortName().sortedCopy(ps)) {
110                 System.out.print(p.getShortName() + "/");
111             }
112             System.out.println();
113 
114             Double score = SelectedPlayers.create(ps, predictions, pcts).score();
115 
116             System.out.println("limit:" + limit + " score:" + score);
117 
118             if (score < bestScore || ps.equals(bestPlayers)) {
119                 return bestPlayers;
120             }
121 
122             bestScore = score;
123             bestPlayers = ps;
124 
125             limit--;
126         }
127 
128         return bestPlayers;
129     }

131     private ImmutableSet<Player> fill(ImmutableSet<Player> partial, ImmutableSet<Player> available) {
132         if (partial.size() == getTargetSize()) {
133             return partial;
134         }
135 
136         AllLineups lineups = new LineupSelection(predictions).dontRequireBackupCatcher().select(partial);
137 
138         Bench bench = Bench.select(lineups, partial, predictions, available, getTargetSize());
139 
140         return ImmutableSet.copyOf(Iterables.concat(partial, bench.players()));
141     }

143     private ImmutableSet<Player> limit(ImmutableSet<Player> best, ImmutableSet<Player> forced, Integer size) {
144 
145         Set<Player> selected = Sets.newHashSet(best);
146 
147         while (selected.size() > size) {
148             AllLineups lineups = new LineupSelection(predictions).dontRequireBackupCatcher().select(selected);
149             Set<Player> ps = Sets.newHashSet(lineups.getAllPlayers());
150 
151             Iterables.removeAll(ps, forced);
152 
153             if (ps.isEmpty()) {
154                 break;
155             }
156 
157             System.out.print("Selected:");
158             for (Player p : byValueProvided(lineups, Iterables.concat(forced, ps)).reverse().sortedCopy(selected)) {
159                 System.out.print(p.getShortName() + "-" + Math.round(getValueProvided(p, lineups, Iterables.concat(forced, ps))) + "/");
160             }
161             System.out.println();
162 
163             selected.remove(byValueProvided(lineups, selected).min(ps));
164         }
165 
166         AllLineups lineups = new LineupSelection(predictions).dontRequireBackupCatcher().select(selected);
167 
168         System.out.print("Limited:");
169         for (Player p : byValueProvided(lineups, selected).reverse().sortedCopy(selected)) {
170             System.out.print(p.getShortName() + "-" + Math.round(getValueProvided(p, lineups, selected)) + "/");
171         }
172         System.out.println();
173 
174         return ImmutableSet.copyOf(selected);
175     }

177     private ImmutableSet<Player> selectStarters(Iterable<Player> ps) {
178 		StarterSelection starters = new StarterSelection(predictions);
179 
180         starters.dontRequireBackupCatcher();
181 
182         return ImmutableSet.copyOf(
183             Iterables.concat(
184                 starters.selectWithDh(Lineup.VsHand.VS_LHP, ps),
185                 starters.selectWithDh(Lineup.VsHand.VS_RHP, ps)));
186     }

188 	private Ordering<Player> byOverall() {
189 		return Ordering
190 			.natural()
191 			.reverse()
192 			.onResultOf(value)
193 			.compound(Player.byTieBreak());
194 	}
195 
196     private Ordering<Player> byValueProvided(final AllLineups lineups, final Iterable<Player> selected) {
197         return Ordering
198             .natural()
199             .onResultOf(new Function<Player, Double>() {
200                 public Double apply(Player p) {
201                     return getValueProvided(p, lineups, selected);
202                 }
203             })
204             .compound(Player.byTieBreak());
205     }
206 
207     private Double getValueProvided(Player p, AllLineups lineups, Iterable<Player> selected) {
208         Double score = 0.0;
209 
210         score += getValueProvided(p, lineups.getVsLhpPlusDh(), selected, pcts.getVsLhpPercentage(), Lineup.VsHand.VS_LHP);
211         score += getValueProvided(p, lineups.getVsLhp(), selected, pcts.getVsLhpPercentage(), Lineup.VsHand.VS_LHP);
212 
213         score += getValueProvided(p, lineups.getVsRhpPlusDh(), selected, pcts.getVsRhpPercentage(), Lineup.VsHand.VS_RHP);
214         score += getValueProvided(p, lineups.getVsRhp(), selected, pcts.getVsRhpPercentage(), Lineup.VsHand.VS_RHP);
215 
216         return score;
217     }

219     private Double getValueProvided(Player p, Lineup l, Iterable<Player> selected, Double pct, Lineup.VsHand vs) {
220         Double score = 0.0;
221         Integer wobaPlus = vs.getStats(predictions, p).getWobaPlus();
222 
223         if (l.contains(p)) {
224             score += wobaPlus;
225 
226             Position pos = l.getPosition(p);
227             score += (Defense.getPositionFactor(pos) * p.getDefensiveRatings().getPositionScore(pos));
228 
229             /*if (p.canPlay(l.getPosition(p))) {
230                 score += wobaPlus;
231             }*/
232         }
233 
234         return pct * score;
235     }
