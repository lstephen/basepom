54      private SimpleRegression getRegression(Predicting predicting) {
55        return getRegression(regressions, predicting);

58      private SimpleRegression getRegression(Map<Predicting, SimpleRegression> r, Predicting p) {
59        if (!r.containsKey(p)) {
60          r.put(p, new SimpleRegression());
61        }
62        return r.get(p);
63      }

65      public Boolean isEmpty() {
66          return getRegression(Predicting.HITS).getN() == 0;
67      }

69      private void addData(TeamStats<BattingStats> teamStats) {
70         for (Player p : teamStats.getPlayers()) {
71             Splits<BattingStats> stats = teamStats.getSplits(p);
72             Splits<BattingRatings<?>> ratings = p.getBattingRatings();
73  
74             addData(regressions, stats.getVsLeft(), ratings.getVsLeft());
75             addData(regressions, stats.getVsRight(), ratings.getVsRight());
76         }
77      }
78  
79      private void addData(Map<Predicting, SimpleRegression> r, BattingStats stats, BattingRatings<?> ratings) {
80          for (int i = 0; i < stats.getPlateAppearances(); i++) {
81              getRegression(r, Predicting.HITS).addData(
82                  ratings.getContact(), stats.getHitsPerPlateAppearance());
83              getRegression(r, Predicting.EXTRA_BASE_HITS).addData(
84                  ratings.getGap(), stats.getExtraBaseHitsPerPlateAppearance());
85  
86              getRegression(r, Predicting.HOME_RUNS).addData(
87                  ratings.getPower(), stats.getHomeRunsPerPlateAppearance());
88              getRegression(r, Predicting.WALKS).addData(ratings.getEye(), stats.getWalksPerPlateAppearance());
89  
90              if (ratings.getK().isPresent()) {
91                  getRegression(r, Predicting.KS).addData(ratings.getK().get(), stats.getKsPerPlateAppearance());
92              }
93          }
94      }

96      private double predict(Predicting predicting, int rating) {
97          return Math.max(0, getRegression(predicting).predict(rating));
98      }

100     public TeamStats<BattingStats> predict(Iterable<Player> ps) {
101         Map<Player, SplitStats<BattingStats>> results = Maps.newHashMap();
102 
103         for (Player p : ps) {
104             if (p.getBattingRatings() != null) {
105                 results.put(p, predict(p));
106             }
107         }
108 
109         return TeamStats.create(results);
110     }

112     public SplitStats<BattingStats> predict(Player p) {
113         return predict(
114             p.getBattingRatings(),
115             !ignoreStatsInPredictions && stats.contains(p)
116                 ? Optional.of(stats.getSplits(p))
117                 : Optional.<SplitStats<BattingStats>>absent());
118     }

120     private SplitStats<BattingStats> predict(
121         Splits<? extends BattingRatings<?>> ratings,
122         Optional<SplitStats<BattingStats>> stats) {
123 
124         Long paToProject = Math.round(DEFAULT_PLATE_APPEARANCES + DEFAULT_PLATE_APPEARANCES * woba.getRSquare());
125 
126         Long vsRightPa = Math.round(paToProject * SplitPercentagesHolder.get().getVsRhpPercentage());
127         Long vsLeftPa = paToProject - vsRightPa;
128 
129         BattingStats vsLeft = predict(ratings.getVsLeft(), vsLeftPa * 100);
130         BattingStats vsRight = predict(ratings.getVsRight(), vsRightPa * 100);
131 
132         if (stats.isPresent()) {
133             SplitStats<BattingStats> splits = stats.get();
134 
135             vsLeft = vsLeft.add(splits.getVsLeft().multiply(100.0));
136             vsRight = vsRight.add(splits.getVsRight().multiply(100.0));
137         }
138 
139         return SplitStats.create(vsLeft, vsRight);
140     }

142     public TeamStats<BattingStats> predictFuture(Iterable<Player> ps) {
143         Map<Player, SplitStats<BattingStats>> results = Maps.newHashMap();
144 
145         for (Player p : ps) {
146             if (p.getBattingRatings() != null) {
147                 SplitStats<BattingStats> prediction =
148                     SplitStats.create(
149                         predict(p.getBattingPotentialRatings().getVsLeft()),
150                         predict(p.getBattingPotentialRatings().getVsRight()));
151 
152                 results.put(p, prediction);
153             }
154         }
155 
156         return TeamStats.create(results);
157     }

159     public SplitStats<BattingStats> predict(Splits<? extends BattingRatings<?>> ratings) {
160         Long paToProject = DEFAULT_PLATE_APPEARANCES;
161 
162         Long vsRightPa = Math.round(paToProject * SplitPercentagesHolder.get().getVsRhpPercentage());
163         Long vsLeftPa = paToProject - vsRightPa;
164 
165         return SplitStats.create(
166             predict(ratings.getVsLeft(), vsLeftPa * 100),
167             predict(ratings.getVsRight(), vsRightPa * 100));
168     }

170     public BattingStats predict(BattingRatings<?> ratings) {
171         return predict(ratings, DEFAULT_PLATE_APPEARANCES);
172     }

174     public BattingStats predict(
175         BattingRatings<?> ratings, Long plateAppearances) {
176 
177         long predictedHits =
178             Math.round(plateAppearances
179                 * predict(Predicting.HITS, ratings.getContact()));
180 
181         long predictedExtraBaseHits =
182             Math.round(plateAppearances
183                 * predict(Predicting.EXTRA_BASE_HITS, ratings.getGap()));
184 
185         double triplesPercentage = (double) leagueBatting.getTriples()
186             / (leagueBatting.getDoubles() + leagueBatting.getTriples());
187         long predictedTriples = Math.round(predictedExtraBaseHits * triplesPercentage);
188         long predictedDoubles = predictedExtraBaseHits - predictedTriples;
189 
190         long predictedHomeRuns =
191              Math.round(plateAppearances
192                 * predict(Predicting.HOME_RUNS, ratings.getPower()));
193 
194         long predictedWalks =
195             Math.round(plateAppearances
196                 * predict(Predicting.WALKS, ratings.getEye()));
197 
198         long predictedKs = ratings.getK().isPresent()
199             ? Math.round(plateAppearances
200                 * predict(Predicting.KS, ratings.getK().get()))
201             : 0;
202 
203         BattingStats predicted = new BattingStats();
204         predicted.setLeagueBatting(leagueBatting);
205         predicted.setAtBats((int) (plateAppearances - predictedWalks));
206         predicted.setHits(predictedHits);
207         predicted.setDoubles(predictedDoubles);
208         predicted.setTriples(predictedTriples);
209         predicted.setHomeRuns(predictedHomeRuns);
210         predicted.setWalks(predictedWalks);
211         predicted.setKs(predictedKs);
212 
213         return predicted;
214     }

220     private void runRegression(Site site) {
221         addData(stats);
222 
223         History history = History.create();
224 
225         int currentSeason = site.getDate().getYear();
226 
227         history.saveBatting(stats, site, currentSeason);
228 
229         Iterable<TeamStats<BattingStats>> historical =
230             history.loadBatting(site, currentSeason, 5);
231 
232         for (TeamStats<BattingStats> h : historical) {
233             addData(h);
234         }
235 
236         Iterable<TeamStats<BattingStats>> all =
237             Iterables.concat(ImmutableList.of(stats), historical);
238 
239         for (TeamStats<BattingStats> tss : all) {
240             for (Player p : tss.getPlayers()) {
241                 SplitStats<BattingStats> splits = tss.getSplits(p);
242                 SplitStats<BattingStats> predicted = predict(p.getBattingRatings());
243 
244                 for (int i = 0; i < splits.getVsLeft().getPlateAppearances(); i++) {
245                     woba.addData(splits.getVsLeft().getWoba(), predicted.getVsLeft().getWoba());
246                 }
247                 for (int i = 0; i < splits.getVsRight().getPlateAppearances(); i++) {
248                     woba.addData(splits.getVsRight().getWoba(), predicted.getVsRight().getWoba());
249                 }
250             }
251         }
252     }

265     public static final class CorrelationReport implements Printable {
266 
267         private final BattingRegression regression;
268 
269         private CorrelationReport(@Nonnull BattingRegression regression) {
270             this.regression = regression;
271         }
272 
273         @Override
274         public void print(PrintWriter w) {
275             w.println("  |   H%  |  XB%  |  HR%  |  BB%  |   K%  |  wOBA |");
276 
277             print(w, "R2", regression.regressions);
278         }
279 
280         private void print(PrintWriter w, String label, Map<Predicting, SimpleRegression> r) {
281             if (r.isEmpty()) { return; }
282 
283             w.println(
284                 String.format(
285                 "%2s| %.3f | %.3f | %.3f | %.3f | %.3f | %.3f |",
286                 label,
287                 r.get(Predicting.HITS).getRSquare(),
288                 r.get(Predicting.EXTRA_BASE_HITS).getRSquare(),
289                 r.get(Predicting.HOME_RUNS).getRSquare(),
290                 r.get(Predicting.WALKS).getRSquare(),
291                 r.get(Predicting.KS).getRSquare(),
292                 regression.woba.getRSquare()));
293         }
294 
295         public static CorrelationReport create(BattingRegression regression) {
296             return new CorrelationReport(regression);
297         }
