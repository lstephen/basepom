1   package com.github.lstephen.ootp.ai.draft;
2   
3   import com.github.lstephen.ootp.ai.player.Player;
4   import com.github.lstephen.ootp.ai.value.TradeValue;
5   import java.io.PrintWriter;
6   import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
7   
8   /**
9    *
10   * @author lstephen
11   */
12  public final class RoundValue {
13  
14    private final TradeValue value;
15  
16    private final DescriptiveStatistics overall = new DescriptiveStatistics();
17  
18    private final DescriptiveStatistics acquisition = new DescriptiveStatistics();
19  
20    private final DescriptiveStatistics historicalOverall = new DescriptiveStatistics();
21  
22    private final DescriptiveStatistics historicalAcquisition = new DescriptiveStatistics();
23  
24    private RoundValue(TradeValue value) {
25      this.value = value;
26    }
27  
28    public boolean isEmpty() {
29      return overall.getN() == 0 && historicalOverall.getN() == 0;
30    }
31  
32    public void add(Player p) {
33      overall.addValue(value.getOverall(p));
34      acquisition.addValue(value.getTradeTargetValue(p));
35      addHistorical(p);
36    }
37  
38    public void add(Iterable<Player> ps) {
39      ps.forEach(this::add);
40    }
41  
42    public void addHistorical(Player p) {
43      historicalOverall.addValue(value.getOverall(p));
44      historicalAcquisition.addValue(value.getTradeTargetValue(p));
45    }
46  
47    public void addHistorical(Iterable<Player> ps) {
48      ps.forEach(this::addHistorical);
49    }
50  
51    private DescriptiveStatistics getOverallStats() {
52      return overall;
53    }
54  
55    private DescriptiveStatistics getAcquisitionStats() {
56      return acquisition;
57    }
58  
59    public static RoundValue create(TradeValue value) {
60      return new RoundValue(value);
61    }
62  
63    public void print(PrintWriter w, String label) {
64      w.print(
65          String.format(
66            "%2s | %3.0f (%3.0f-%3.0f) | %3.0f (%3.0f-%3.0f) %3.0f/%3.0f | %3d/%3d |",
67            label,
68            getOverallStats().getPercentile(50),
69            getOverallStats().getMin(),
70            getOverallStats().getMax(),
71            getAcquisitionStats().getPercentile(50),
72            getAcquisitionStats().getMin(),
73            getAcquisitionStats().getMax(),
74            getAcquisitionStats().getPercentile(50) / 1.1,
75            getAcquisitionStats().getPercentile(50) * 1.1,
76            getOverallStats().getN(),
77            getAcquisitionStats().getN()));
78  
79      w.print(
80          String.format(
81            "| %3.0f (%3.0f-%3.0f) | %3.0f (%3.0f-%3.0f) %3.0f/%3.0f | %3d/%3d",
82            historicalOverall.getPercentile(50),
83            historicalOverall.getMin(),
84            historicalOverall.getMax(),
85            historicalAcquisition.getPercentile(50),
86            historicalAcquisition.getMin(),
87            historicalAcquisition.getMax(),
88            historicalAcquisition.getPercentile(50) / 1.1,
89            historicalAcquisition.getPercentile(50) * 1.1,
90            historicalOverall.getN(),
91            historicalAcquisition.getN()));
92      w.println();
93      w.flush();
94    }
95  
96  
97  
98  }
