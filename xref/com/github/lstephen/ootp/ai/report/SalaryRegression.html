1   package com.github.lstephen.ootp.ai.report;
2   
3   import com.google.common.base.Strings;
4   import com.github.lstephen.ootp.ai.io.Printable;
5   import com.github.lstephen.ootp.ai.player.Player;
6   import com.github.lstephen.ootp.ai.site.Site;
7   import com.github.lstephen.ootp.ai.value.SalaryPredictor;
8   import com.github.lstephen.ootp.ai.value.TradeValue;
9   import java.io.PrintWriter;
10  import org.apache.commons.math3.stat.regression.SimpleRegression;
11  
12  /**
13   *
14   * @author lstephen
15   */
16  public class SalaryRegression implements SalaryPredictor, Printable {
17  
18      private final TradeValue trade;
19  
20      private final Site site;
21  
22      private final SimpleRegression vsOverall = new SimpleRegression();
23  
24      private final SimpleRegression vsReplacement = new SimpleRegression(false);
25  
26      private final SimpleRegression vsAcquisition = new SimpleRegression();
27  
28      public SalaryRegression(TradeValue trade, Site site) {
29          this.trade = trade;
30          this.site = site;
31      }
32  
33      private Integer leagueMinimum() {
34          if (site.getName().equals("BTH")) {
35              return 500000;
36          }
37          if (site.getName().equals("PSD")) {
38              return 248045;
39          }
40          return 300000;
41      }
42  
43      public void add(Player p) {
44          if (p != null
45              && !Strings.isNullOrEmpty(p.getSalary())
46              && !p.getSalary().endsWith("a")
47              && !p.getSalary().endsWith("r")) {
48  
49              addData(p, site.getCurrentSalary(p) - leagueMinimum());
50          }
51      }
52  
53      public void add(Iterable<Player> ps) {
54          for (Player p : ps) {
55              add(p);
56          }
57      }
58  
59      private void addData(Player p, Integer salary) {
60          vsOverall.addData(trade.getOverall(p), salary);
61          vsReplacement.addData(trade.getCurrentValueVsReplacement(p), salary);
62          vsAcquisition.addData(trade.getTradeTargetValue(p), salary);
63      }
64  
65      public Integer predict(Player p) {
66          Integer salary  = Math.max(
67              0,
68              (int) vsReplacement.predict(trade.getCurrentValueVsReplacement(p)) + leagueMinimum());
69  
70          return salary > 0 && salary < leagueMinimum()
71              ? leagueMinimum()
72              : salary;
73      }
74  
75      private Integer getErrorRange(Player p) {
76          Double pctError = Math.pow(1.0 - vsReplacement.getRSquare(), 2);
77  
78          return (int) (predict(p) * pctError);
79      }
80  
81      public Integer predictMinimum(Player p) {
82          Integer min = predict(p) - getErrorRange(p);
83  
84          return min > 0 && min < leagueMinimum()
85              ? leagueMinimum()
86              : min;
87      }
88  
89      public Integer predictMaximum(Player p) {
90          Integer max = predict(p) + getErrorRange(p);
91  
92          return max > 0 && max < leagueMinimum()
93              ? leagueMinimum()
94              : max;
95      }
96  
97      @Override
98      public void print(PrintWriter w) {
99  
100         w.println();
101 
102         w.println("  | vsOvr | vsRpl | vsAcq |");
103         w.println(
104             String.format(
105                 "R2| %-9.3f | %-9.3f | %-9.3f |",
106                 vsOverall.getRSquare(),
107                 vsReplacement.getRSquare(),
108                 vsAcquisition.getRSquare()));
109         w.println(
110             String.format(
111                 "R2| %9.3e | %9.3e | %9.3e |",
112                 vsOverall.getMeanSquareError(),
113                 vsReplacement.getMeanSquareError(),
114                 vsAcquisition.getMeanSquareError()));
115     }
116 
117 }
