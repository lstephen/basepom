104     private Set<TeamScore> calculateScores() {
105         Set<TeamScore> scores = Sets.newHashSet();
106 
107         for (Id<Team> id : site.getTeamIds()) {

118     @Override
119     public void print(PrintWriter w) {
120         w.println();
121         w.println(String.format("**%-18s | %-5s %-5s %-5s | %-5s %-5s %-5s | (rpg:%.2f)", title, " Bat", " LU", " Ovr", " Pit", " Rot", " Ovr", getExpectedRunsPerGame()));
122 
123         for (LeagueStructure.League league : site.getLeagueStructure().getLeagues()) {
124             w.println();
125             w.println(league.getName());
126 
127             for (LeagueStructure.Division division : league.getDivisions()) {
128                 if (!Strings.isNullOrEmpty(division.getName())) {
129                     w.println(StringUtils.rightPad(division.getName() + " ", 21, '-') + "+");
130                 }
131 
132                 Set<TeamScore> divisionScores = Sets.newHashSet();
133 
134                 for (Id<Team> team : division.getTeams()) {
135                     divisionScores.add(getScore(team));
136                 }
137 
138                 for (TeamScore s : getOrdering().sortedCopy(divisionScores)) {
139                     printTeamScore(s, w);
140                 }
141             }
142         }
143 
144         w.println(String.format("%-20s | %11s %5.1f | %11s %5.1f |", "", "", mlHitting.getMean(), "", mlPitching.getMean()));
145 
146         w.flush();
147     }

155     private void printTeamScore(TeamScore s, PrintWriter w) {
156         Record current = site.getStandings().getRecord(s.getId());

164                 s.getLineup(),

166                 s.getPitching(),

182     public Record getExpectedEndOfSeason(Id<Team> team) {
183         Standings standings = site.getStandings();
184 
185         Record current = standings.getRecord(team);
186 
187         Long eosWs = current.getWins()
188             + Math.round(
189                 getScore(team)
190                     .getExpectedWinningPercentage(getExpectedRunsPerGame())
191                     * (162 - current.getGames()));
192 
193         Long eosLs = 162 - eosWs;
194 
195         return Record.create(eosWs, eosLs);
196     }

198     private Set<TeamScore> normalize(Iterable<TeamScore> scores) {
199         Double battingAverage = getBattingAverage(scores);
200         Double lineupAverage = getLineupAverage(scores);
201         Double pitchingAverage = getPitchingAverage(scores);
202         Double rotationAverage = getRotationAverage(scores);
203 
204         Set<TeamScore> normalized = Sets.newHashSet();
205 
206         for (TeamScore s : scores) {
207             normalized.add(
208                 TeamScore.create(
209                     s.getId(),
210                     normalize(s.getBatting(), battingAverage),
211                     normalize(s.getLineup(), lineupAverage),
212                     normalize(s.getPitching(), pitchingAverage),
213                     normalize(s.getRotation(), rotationAverage)));
214         }
215 
216         return normalized;
217     }
218 
219     private Double normalize(Double value, Double average) {
220         return value * 100 / average;

276     private Double calculateLineup(Iterable<Player> players) {
277         return calculateScore(
278             SlotSelection
279                 .builder()
280                 .ordering(Ordering.natural().reverse().onResultOf(value))
281                 .slots(ImmutableMultiset.of(Slot.C, Slot.SS, Slot.IF, Slot.IF, Slot.CF, Slot.OF, Slot.OF, Slot.H, Slot.H))
282                 .size(9)
283                 .fillToSize(Slot.H)
284                 .build(),
285             Selections.onlyHitters(players));
286     }

294     private Double calculateRotation(Iterable<Player> players) {
295         return calculateScore(
296             SlotSelection
297                 .builder()
298                 .ordering(Ordering.natural().reverse().onResultOf(value))
299                 .slots(ImmutableMultiset.of(Slot.SP, Slot.SP, Slot.SP, Slot.SP, Slot.MR, Slot.MR, Slot.MR))
300                 .size(7)
301                 .fillToSize(Slot.P)
302                 .build(),
303             Selections.onlyPitchers(players));

306     private Double calculateScore(Selection selection, Iterable<Player> players) {
307         Iterable<Player> ml = selection
308             .select(ImmutableSet.<Player>of(), players)
309             .values();
310 
311         DescriptiveStatistics stats = new DescriptiveStatistics();
312 
313         for (Player p : ml) {
314             stats.addValue(value.apply(p));
315             addToAverages(p);
316         }
317 
318         return stats.getMean();
319     }
320 
321     private void addToAverages(Player p) {
322         if (Selections.isHitter(p)) {
323             mlHitting.addValue(value.apply(p));
324         }
325 
326         if (Selections.isPitcher(p)) {
327             mlPitching.addValue(value.apply(p));
328         }
329     }

335     private static final class TeamScore {
336 
337         private final Id<Team> id;
338 
339         private final Double batting;
340 
341         private final Double lineup;
342 
343         private final Double pitching;
344 
345         private final Double rotation;
346 
347         private TeamScore(Id<Team> id, Double batting, Double lineup, Double pitching, Double rotation) {
348             this.id = id;
349             this.batting = batting;
350             this.lineup = lineup;
351             this.pitching = pitching;
352             this.rotation = rotation;
353         }
354 
355         public Id<Team> getId() {
356             return id;
357         }
358 
359         public Double getBatting() {
360             return batting;
361         }
362 
363         public Double getLineup() {
364             return lineup;
365         }
366 
367         public Double getPitching() {
368             return pitching;
369         }
370 
371         public Double getRotation() {
372             return rotation;
373         }
374 
375         public Double getOverallBatting() {
376             return combine(batting, lineup);
377         }
378 
379         public Double getOverallPitching() {
380             return combine(pitching, rotation);
381         }
382 
383         private Double combine(Double depth, Double strength) {
384             double a = .5 * depth / 100;
385             double b = .5 * 100 / strength;
386 
387             double log5 = (a - a*b) / (a + b - 2*a*b);
388 
389             return 100 * log5 / .5;
390         }
391 
392         /**
393          * Assumption - normalized to an average of 100
394          * @return
395          */
396         public Double getExpectedWinningPercentage(double rpg) {
397             int averageRuns = (int) (rpg * 162);
398 
399             double rs = getOverallBatting() * averageRuns / 100;
400             double ra = (100.0 / getOverallPitching() * 100) * averageRuns / 100;
401 
402             return 1.0 / (1.0 + Math.pow(ra / rs, 2));
403         }
404 
405         public String getExpectedReocrd(double rpg) {
406             Long wins = Math.round(162 * getExpectedWinningPercentage(rpg));
407 
408             return String.format("%3d-%3d", wins, 162-wins);
409         }
410 
411         public static TeamScore create(
412             Id<Team> id, Double batting, Double lineup, Double pitching, Double rotation) {
413 
414             return new TeamScore(id, batting, lineup, pitching, rotation);
415         }
416 
417         public static Ordering<TeamScore> byWinningPercentage(final double rpg) {
418             return Ordering
419                 .natural()
420                 .reverse()
421                 .onResultOf(s -> s.getExpectedWinningPercentage(rpg));
422         }

424     }
